# An example test.t Test::Base test in Perl
use Test::Base tests => 2;
use JSON::XS;

filters { perl => [qw(eval encode)] };
run_is perl => 'json';

sub encode {
    return encode_json(shift) . "\n";
}

__END__
=== Simple hash
--- perl
{ size => 42, color => 'green' }
--- json
{"color":"green","size":42}

=== Simple array
--- perl
[qw(red white blue)]
--- json
["red","white","blue"]
